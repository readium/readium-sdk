59 bool Container::Open(const string& path)
60 {
61  _archive = Archive::Open(path.stl_str());
62  if (_archive == nullptr)
63  throw std::invalid_argument(_Str("Path does not point to a recognised archive file: '", path, "'"));
64  _path = path;
65 
66  // TODO: Initialize lazily? Doing so would make initialization faster, but require
67  // PackageLocations() to become non-const, like Packages().
68  ArchiveXmlReader reader(_archive->ReaderAtPath(gContainerFilePath));
69  if (!reader) {
70  throw std::invalid_argument(_Str("Path does not point to a recognised archive file: '", path, "'"));
71  }
72 #if EPUB_USE(LIBXML2)
73  _ocf = reader.xmlReadDocument(gContainerFilePath, nullptr, XML_PARSE_RECOVER|XML_PARSE_NOENT|XML_PARSE_DTDATTR);
74 #else
75  decltype(_ocf) __tmp(reader.ReadDocument(gContainerFilePath, nullptr, /*RESOLVE_EXTERNALS*/ 1));
76  _ocf = __tmp;
77 #endif
78  if (!((bool)_ocf))
79  return false;
80 
81 #if EPUB_COMPILER_SUPPORTS(CXX_INITIALIZER_LISTS)
82  XPathWrangler xpath(_ocf, { { "ocf", "urn:oasis:names:tc:opendocument:xmlns:container" } });
83 #else
84  XPathWrangler::NamespaceList __ns;
85  __ns["ocf"] = OCFNamespaceURI;
86  XPathWrangler xpath(_ocf, __ns);
87 #endif
88  xml::NodeSet nodes = xpath.Nodes(gRootfilesXPath);
89 
90  if (nodes.empty())
91  return false;
92 
93  LoadEncryption();
94 
95  ParseVendorMetadata();
96 
97  for (auto n : nodes)
98  {
99  string type = _getProp(n, "media-type");
100 
101  string path = _getProp(n, "full-path");
102  if (path.empty())
103  continue;
104 
105  auto pkg = Package::New(Ptr(), type);
106  if (pkg->Open(path))
107  _packages.push_back(pkg);
108  }
109 
110  auto fm = FilterManager::Instance();
111  for (auto& pkg : _packages)
112  {
113  auto fc = fm->BuildFilterChainForPackage(pkg);
114  pkg->SetFilterChain(fc);
115  }
116 
117  return true;
118 }

224 void Container::ParseVendorMetadata()
225 {
226  unique_ptr<ArchiveReader> pZipReader = _archive->ReaderAtPath(gAppleiBooksDisplayOptionsFilePath);
227  if ( !pZipReader )
228  return;
229 
230  ArchiveXmlReader reader(std::move(pZipReader));
231 #if EPUB_USE(LIBXML2)
232  shared_ptr<xml::Document> docXml = reader.xmlReadDocument(gAppleiBooksDisplayOptionsFilePath, nullptr, XML_PARSE_RECOVER|XML_PARSE_NOENT|XML_PARSE_DTDATTR);
233 #elif EPUB_USE(WIN_XML)
234  auto docXml = reader.ReadDocument(gAppleiBooksDisplayOptionsFilePath, nullptr, 0);
235 #endif
236  if ( !bool(docXml) )
237  return;
238 
239 #if EPUB_COMPILER_SUPPORTS(CXX_INITIALIZER_LISTS)
240  XPathWrangler xpath(docXml);
241 #else
242  XPathWrangler::NamespaceList __ns;
243  XPathWrangler xpath(docXml, __ns);
244 #endif
245 
246  xml::NodeSet nodes = xpath.Nodes("/display_options/platform/option");
247  if ( nodes.empty() )
248  {
249  //xml::string str(docXml->XMLString());
250  //printf("%s\n", docXml->XMLString().utf8());
251  return;
252  }
253 
254  for ( auto node : nodes )
255  {
256  string name = _getProp(node, "name");
257  if (name.empty())
258  continue;
259 
260  if (name == "fixed-layout")
261  {
262  _appleIBooksDisplayOption_FixedLayout = node->Content(); // true | false
263  }
264  else if (name == "orientation-lock")
265  {
266  _appleIBooksDisplayOption_Orientation = node->Content(); // landscape-only | portrait-only | none
267  }
268  }
269 }

271 void Container::LoadEncryption()
272 {
273  unique_ptr<ArchiveReader> pZipReader = _archive->ReaderAtPath(gEncryptionFilePath);
274  if ( !pZipReader )
275  return;
276 
277  ArchiveXmlReader reader(std::move(pZipReader));
278 #if EPUB_USE(LIBXML2)
279  shared_ptr<xml::Document> enc = reader.xmlReadDocument(gEncryptionFilePath, nullptr, XML_PARSE_RECOVER|XML_PARSE_NOENT|XML_PARSE_DTDATTR);
280 #elif EPUB_USE(WIN_XML)
281  auto enc = reader.ReadDocument(gEncryptionFilePath, nullptr, 0);
282 #endif
283  if ( !bool(enc) )
284  return;
285 #if EPUB_COMPILER_SUPPORTS(CXX_INITIALIZER_LISTS)
286  XPathWrangler xpath(enc, {{"enc", XMLENCNamespaceURI}, {"ocf", OCFNamespaceURI}});
287 #else
288  XPathWrangler::NamespaceList __ns;
289  __ns["ocf"] = OCFNamespaceURI;
290  __ns["enc"] = XMLENCNamespaceURI;
291  XPathWrangler xpath(enc, __ns);
292 #endif
293  xml::NodeSet nodes = xpath.Nodes("/ocf:encryption/enc:EncryptedData");
294  if ( nodes.empty() )
295  {
296  xml::string str(enc->XMLString());
297  printf("%s\n", enc->XMLString().utf8());
298  return; // should be a hard error?
299  }
300 
301  for ( auto node : nodes )
302  {
303  auto encPtr = EncryptionInfo::New(Ptr());
304  if ( encPtr->ParseXML(node) )
305  _encryption.push_back(encPtr);
306  }
307 }

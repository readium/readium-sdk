29 static void RemoveFragment(string_view& str)
30 {
31  auto idx = str.rfind('#');
32  if (idx != string_view::npos)
33  str.remove_suffix(str.size()-idx);
34 }
35 static bool MatchesWithoutFragment(string_view a, string_view b)
36 {
37  RemoveFragment(a);
38  RemoveFragment(b);
39  return a == b;
40 }

97 ConstManifestItemPtr Link::ReferencedManifestItem() const
98 {

102

104  {
105  // remember: <collection> is within the same OPF file as the manifest, so

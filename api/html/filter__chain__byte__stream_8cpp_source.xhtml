71 ByteStream::size_type FilterChainByteStream::ReadBytes(void* bytes, size_type len)
72 {
73  if (len == 0) return 0;
74 
75  if (_needs_cache)
76  {
77  if (_cache.GetBufferSize() == 0 && _input->AtEnd() == false)
78  CacheBytes();
79 
80  return ReadBytesFromCache(bytes, len);
81  }
82 
83  if (_read_cache.GetBufferSize() > 0)
84  {
85  size_type toMove = std::min(len, _read_cache.GetBufferSize());
86  ::memcpy_s(bytes, len, _read_cache.GetBytes(), toMove);
87  _read_cache.RemoveBytes(toMove);
88  return toMove;
89  }
90  else
91  {
92  if (!_input->IsOpen())
93  {
94  return 0;
95  }
96

98  if (result == 0)
99  {
100  return 0;
101  }
102 
103  result = FilterBytes(bytes, result);
104  if (result == 0)
105  {
106  return 0;
107  }
108 
109  size_type toMove = std::min(len, _read_cache.GetBufferSize());
110  if (toMove > 0)
111  {
112  ::memcpy_s(bytes, len, _read_cache.GetBytes(), toMove);
113  _read_cache.RemoveBytes(toMove);
114  }
115  return toMove;
116  }

119 ByteStream::size_type FilterChainByteStream::FilterBytes(void* bytes, size_type len)
120 {
121  if (len == 0) return 0;
122 
123  size_type result = len;
124  ByteBuffer buf(reinterpret_cast<uint8_t*>(bytes), len);
125  buf.SetUsesSecureErasure();
126 
127  for (int i = 0; i < m_filters.size(); i++)
128  {
129  ContentFilterPtr filter = m_filters.at(i);
130  FilterContext * filterContext = m_filterContexts.at(i).get();
131 
132  size_type streamPos = 0;
133 
134  // A filter may support ranges, but may be invoked in a non-HTTP-byte-range scenario
135  RangeFilterContext *filterContextRange = dynamic_cast<RangeFilterContext *>(filterContext);
136  if (filterContextRange != nullptr)
137  {
138  // ASSERT i == 0 ? (decryption filter always first, can access raw byte stream directly)
139 
140  ByteRange byteRange;
141  if (!_needs_cache)
142  {
143  // ASSERT m_filters.size() == 1 ? (decryption filter on its own might be subject to sequential consecutive byte chunks)
144 
145  if (!_input->IsOpen())
146  {
147  throw std::logic_error("ChainLinkProcessor: Byte stream not open?!");
148  }
149 
150  streamPos = _input->Position();
151 
152  byteRange.Location(streamPos - result);
153  byteRange.Length(result);
154  }
155  filterContextRange->GetByteRange() = byteRange;
156  filterContextRange->SetSeekableByteStream(_input.get());
157  }
158 
159  size_type filteredLen = 0;
160  void *filteredData = nullptr;
161 
162  if (filterContextRange != nullptr)
163  {
164  filteredData = filter->FilterData(filterContext, nullptr, 0, &filteredLen);
165  }
166  else
167  {
168  filteredData = filter->FilterData(filterContext, buf.GetBytes(), buf.GetBufferSize(), &filteredLen);
169  }
170 
171  if (filterContextRange != nullptr)
172  {
173  filterContextRange->GetByteRange().Reset();
174  filterContextRange->ResetSeekableByteStream();
175 
176  if (!_needs_cache)
177  {
178  if (_input->IsOpen())
179  {
180  _input->Seek(streamPos, std::ios::beg);
181  }
182  }
183  }
184 
185  if (filteredData == nullptr || filteredLen == 0)
186  {
187  if (filteredData != nullptr && filteredData != buf.GetBytes())
188  {
189  if (filterContextRange == nullptr || reinterpret_cast<uint8_t*>(filteredData) != filterContextRange->GetCurrentTemporaryByteBuffer())
190  {
191  delete[] reinterpret_cast<uint8_t*>(filteredData);
192  }
193  }
194 
195  return 0;
196  }
197 
198  result = filteredLen;
199 
200  if (filteredData != buf.GetBytes())
201  {
202  // NOTE: destroys previous buffer, allocates new memory block! (memcpy)
203  buf = ByteBuffer(reinterpret_cast<uint8_t*>(filteredData), result);
204 
205  if (filterContextRange == nullptr || reinterpret_cast<uint8_t*>(filteredData) != filterContextRange->GetCurrentTemporaryByteBuffer())
206  {
207  delete[] reinterpret_cast<uint8_t*>(filteredData);
208  }
209  }
210  else if (result < buf.GetBufferSize())
211  {
212  buf.RemoveBytes(buf.GetBufferSize() - result, result);
213 
214  // NOTE: destroys previous buffer, allocates new memory block! (memcpy)
215  //buf = ByteBuffer(reinterpret_cast<uint8_t*>(filteredData), result);
216  }
217  else if (result > buf.GetBufferSize())
218  {
219  // This should never happen! (returned more bytes than could fit!)
220 
221  // NOTE: destroys previous buffer, allocates new memory block! (memcpy)
222  buf = ByteBuffer(reinterpret_cast<uint8_t*>(filteredData), result);
223  }
224  }
225 
226  _read_cache = std::move(buf);
227 
228  // ASSERT result == _read_cache.GetBufferSize()
229  return result;
230 }

242 void FilterChainByteStream::CacheBytes()
243 {
244  // read everything from the input stream
245 #define _TMP_BUF_LEN 16*1024
246  uint8_t buf[_TMP_BUF_LEN] = {};
247  while (_input->AtEnd() == false)
248  {
249  size_type numRead = _input->ReadBytes(buf, _TMP_BUF_LEN);
250  if (numRead == 0)
251  break;
252  if (numRead > 0)
253  _cache.AddBytes(buf, numRead);
254  }
255 
256  if (_cache.GetBufferSize() == 0) return;
257 
258  // filter everything completely
259  size_type filtered = FilterBytes(_cache.GetBytes(), _cache.GetBufferSize());
260 
261  if (filtered > 0)
262  {
263  // ASSERT filtered == _read_cache.GetBufferSize()
264 
265  _cache = std::move(_read_cache);
266  _read_cache.RemoveBytes(_read_cache.GetBufferSize());
267  }
268 
269  // this potentially contains decrypted data, so use secure erasure
270  _cache.SetUsesSecureErasure();
271 }

71 bool XPathEvaluator::Evaluate(std::shared_ptr<const Node> node, ObjectType * resultType)
72 {
73  try
74  {
75  if (_namespaces.empty())
76  {
77  _lastResult = node->xml()->SelectNodes(_xpath);
78  }
79  else
80  {
81  std::wstringstream s;
82  for (auto& pair : _namespaces)
83  {
84  if (pair.second.empty())
85  continue;
86 
87  if (s.tellp() > 0)
88  s << TEXT(' ');
89  s << TEXT("xmlns");
90  if (!pair.first.empty())
91  {
92  s << TEXT(':');
93  s << pair.first;
94  }
95  s << TEXT("=\"");
96  s << pair.second;
97  s << TEXT('"');
98  }
99 
100  string ns(s.str());
101  try
102  {
103  _lastResult = node->xml()->SelectNodesNS(_xpath, ns);
104  }
105  catch (...)
106  {
107  _lastResult = nullptr;
108  }
109  }
110 
111  if (_lastResult == nullptr)
112  return false;
113 
114  if (resultType != nullptr)
115  {
116  if (_lastResult->Size > 1)
117  {
118  *resultType = ObjectType::NodeSet;
119  }
120  else
121  {
122  IXmlNode^ inode = _lastResult->Item(0);
123  using ::Windows::Data::Xml::Dom::NodeType;
124  if ((inode != nullptr) && (inode->NodeType == NodeType::TextNode))
125  {
126  XmlText^ txt = dynamic_cast<XmlText^>(inode);
127  string str(txt->Data);
128  if (_wtoi(str.c_str()) != 0)
129  {
130  *resultType = ObjectType::Number;
131  }
132  else if (::_wcsnicmp(str.c_str(), __TEXT("true"), min((string::size_type)4, str.length())) == 0)
133  {
134  *resultType = ObjectType::Boolean;
135  }
136  else if (::_wcsnicmp(str.c_str(), __TEXT("yes"), min((string::size_type)3, str.length())) == 0)
137  {
138  *resultType = ObjectType::Boolean;
139  }
140  else
141  {
142  *resultType = ObjectType::String;
143  }
144  }
145  else
146  {
147  *resultType = ObjectType::NodeSet;
148  }
149  }
150  }
151 
152  return true;
153  }
154  catch (...)
155  {
156  }
157 
158  return false;

205 bool XPathEvaluator::BooleanResult() const
206 {
207  if (_lastResult == nullptr)
208  throw InternalError(std::string(__PRETTY_FUNCTION__) + " called when no result available");
209  if (_lastResult->Size == 0)
210  return false;
211  IXmlNode^ node = _lastResult->Item(0);
212  if (NodeType(node->NodeType) != NodeType::Text)
213  return true;
214 
215  string str(dynamic_cast<XmlText^>(node)->Data);
216  if (::_wcsnicmp(str.c_str(), __TEXT("true"), min((string::size_type)4, str.length())) == 0)
217  {
218  return true;
219  }
220  else if (::_wcsnicmp(str.c_str(), __TEXT("yes"), min((string::size_type)3, str.length())) == 0)
221  {
222  return true;
223  }
224 
225  return (_wtoi(str.c_str()) != 0);
226 }

166  const bool ClipOffset(uint32_t & offset, shared_ptr<const Parallel> par) const
167  {
168  std::shared_ptr<SMILData> smilData = Owner(); // internally: std::weak_ptr<SMILData>.lock()
169  if (smilData == nullptr)
170  {
171  return false;
172  }
173 
174  for (int i = 0; i < _children.size(); i++)
175  {
176  auto container = _children[i];
177  if (container->IsParallel())
178  {
179  auto para = std::dynamic_pointer_cast<const Parallel>(container);
180  if (para == par)
181  {
182  return true;
183  }
184 
185  if (para->Audio() == nullptr)
186  {
187  continue;
188  }
189 
190  if (para->Text() != nullptr && para->Text()->SrcManifestItem() != nullptr && para->Text()->SrcManifestItem() != smilData->XhtmlSpineItem()->ManifestItem())
191  {
192  continue;
193  }
194 
195  uint32_t clipDur = para->Audio()->ClipDurationMilliseconds();
196  offset += clipDur;
197  }
198  else if (container->IsSequence())
199  {
200  auto sequence = std::dynamic_pointer_cast<const Sequence>(container);
201 
202  bool found = sequence->ClipOffset(offset, par);
203  if (found)
204  {
205  return true;
206  }
207  }
208  }
209 
210  return false;
211  }

213  shared_ptr<const Parallel> ParallelAt(uint32_t timeMilliseconds) const
214  {
215  std::shared_ptr<SMILData> smilData = Owner(); // internally: std::weak_ptr<SMILData>.lock()
216  if (smilData == nullptr)
217  {
218  return nullptr;
219  }
220 
221  uint32_t offset = 0;
222 
223  for (int i = 0; i < _children.size(); i++)
224  {
225  uint32_t timeAdjusted = timeMilliseconds - offset;
226 
227  auto container = _children[i];
228  if (container->IsParallel())
229  {
230  auto para = std::dynamic_pointer_cast<const Parallel>(container);
231 
232  if (para->Audio() == nullptr)
233  {
234  continue;
235  }
236 
237  if (para->Text() != nullptr && para->Text()->SrcManifestItem() != nullptr && para->Text()->SrcManifestItem() != smilData->XhtmlSpineItem()->ManifestItem())
238  {
239  continue;
240  }
241 
242  uint32_t clipDur = para->Audio()->ClipDurationMilliseconds();
243 
244  if (clipDur > 0 && timeAdjusted <= clipDur)
245  {
246  return para;
247  }
248 
249  offset += clipDur;
250  }
251  else if (container->IsSequence())
252  {
253  auto sequence = std::dynamic_pointer_cast<const Sequence>(container);
254 
255  auto para = sequence->ParallelAt(timeAdjusted);
256  if (para != nullptr)
257  {
258  return para;
259  }
260 
261  offset += sequence->DurationMilliseconds();
262  }
263  }
264 
265  return nullptr;
266  }

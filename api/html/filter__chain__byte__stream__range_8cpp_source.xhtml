69 ByteStream::size_type FilterChainByteStreamRange::ReadBytes(void *bytes, size_type len, ByteRange &byteRange)
70 {
71  if (len == 0) return 0;
72 
73  if (byteRange.Length() == 0 && !byteRange.IsFullRange()) // invalid range
74  {
75  return 0;
76  }
77 
78  if (byteRange.Length() > len && !byteRange.IsFullRange()) // too small buffer
79  {
80  return 0;
81  }
82 
83  if (!m_filter)
84  {
85  // There are no ContentFilters that applied. In this case, the caller is just interested
86  // in getting the raw bytes out of the ZIP file. So, then, just read the raw bytes.
87  return ReadRawBytes(bytes, len, byteRange);
88  }
89 
90  RangeFilterContext *filterContext = dynamic_cast<RangeFilterContext *>(m_filterContext.get());
91  if (filterContext != nullptr)
92  {
93  filterContext->GetByteRange() = byteRange;
94  filterContext->SetSeekableByteStream(m_input.get());
95  }
96 
97  size_type filteredLen = 0;
98  void *filteredData = nullptr;
99 
100  if (filterContext != nullptr) {
101  filteredData = m_filter->FilterData(m_filterContext.get(), nullptr, 0, &filteredLen);
102  } else {
103  size_type result = m_input->ReadBytes(bytes, len);
104  if (result == 0) return 0;
105 
106  ByteBuffer buf(reinterpret_cast<uint8_t*>(bytes), result);
107  buf.SetUsesSecureErasure();
108 
109  filteredData = m_filter->FilterData(m_filterContext.get(), buf.GetBytes(), buf.GetBufferSize(), &filteredLen);
110  }
111 
112  if (filterContext != nullptr)
113  {
114  filterContext->GetByteRange().Reset();
115  filterContext->ResetSeekableByteStream();
116  }
117 
118  if (filteredData == nullptr || filteredLen == 0)
119  {
120  if (filteredData != nullptr && filteredData != bytes)
121  {
122  if (filterContext == nullptr || reinterpret_cast<uint8_t*>(filteredData) != filterContext->GetCurrentTemporaryByteBuffer())
123  {
124  delete[] reinterpret_cast<uint8_t *>(filteredData);
125  }
126  }
127 
128  return 0;
129  }
130 
131  if (filteredData != bytes)
132  {
133  ByteStream::size_type toCopy = std::min(len, filteredLen);
134  ::memcpy_s(bytes, len, filteredData, toCopy);
135 
136  if (filteredLen > len)
137  {
138  ByteStream::size_type toCache = filteredLen - len;
139 
140  //TODO cache remainder bytes for use next time round...
141  // (e.g. when decrypted bytes are greater than original request?)
142  }
143 
144  if (filterContext == nullptr || reinterpret_cast<uint8_t*>(filteredData) != filterContext->GetCurrentTemporaryByteBuffer())
145  {
146  delete[] reinterpret_cast<uint8_t *>(filteredData);
147  }
148  }
149 
150  return filteredLen;
151 }

153 ByteStream::size_type FilterChainByteStreamRange::ReadRawBytes(void *bytes, size_type len, ePub3::ByteRange &byteRange)
154 {
155  if (len == 0) return 0;
156 
157  size_type bytesToRead = 0;
158  if (!byteRange.IsFullRange())
159  {
160  m_input->Seek(byteRange.Location(), std::ios::beg);
161  bytesToRead = std::min(len, (size_type)byteRange.Length());
162  }
163  else
164  {
165  m_input->Seek(0, std::ios::beg);
166  if (m_input->BytesAvailable() > len)
167  {
168  return 0; // Buffer is not big enough to take the entire file.
169  }
170  bytesToRead = len;
171  }
172 
173  if (bytesToRead == 0) return 0;
174 
175  return m_input->ReadBytes(bytes, bytesToRead);
176 }

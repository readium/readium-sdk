37 int InputBuffer::read_cb(void *context, char *buffer, int len)
38 {
39  InputBuffer * p = reinterpret_cast<InputBuffer*>(context);
40 
41  size_t toRead = static_cast<size_t>(len);
42  uint8_t* buf = reinterpret_cast<uint8_t*>(buffer);
43 
44  size_t res = 0;
45  if (p->_encodingCheck == "utf-8" && len >= 3)
46  {
47  res = p->read(buf, 3);
48 
49  // BOM check (0xEF,0xBB,0xBF)
50  if (res == 3 && buf[0] == 0xEF && buf[1] == 0xBB && buf[2] == 0xBF)
51  {
52  // Skip BOM bytes
53  res = p->read(buf, toRead - 3);
54  }
55  else if (res > 0)
56  {
57  // no BOM, read more bytes
58  res += p->read(buf + res, toRead - res);
59  }
60  }
61  else
62  {
63  res = p->read(buf, toRead);
64  }
65  p->_encodingCheck = NULL;
66 
67  return static_cast<int>(res);
68 }

67 future<ContainerPtr>
68 ContentModuleManager::LoadContentAtPath(const string& path, launch policy)
69 {
70  std::unique_lock<std::mutex>(_mutex);
71 
72  if (_known_modules.empty())
73  {
74  // special case for when we don't have any Content Modules to rely on for an initialized result
75  return make_ready_future<ContainerPtr>(ContainerPtr(nullptr));
76  }
77 
78  future<ContainerPtr> result;
79  for (auto& item : _known_modules)
80  {
81  auto modulePtr = item.second;
82  result = modulePtr->ProcessFile(path, policy);
83 
84  // check the state of the future -- has it already been set?
85  future_status status = result.wait_for(std::chrono::system_clock::duration(0));
86 
87  // if it's ready, the call to get() will never block
88  if (status == future_status::ready) {
89  // unpack the future
90  ContainerPtr container = result.get();
91 
92  if (bool(container)) {
93  // we have a valid container already
94  result = make_ready_future(container);
95  result = result.then([modulePtr](future<ContainerPtr> fut) {
96  ContainerPtr ptr = fut.get();
97  modulePtr->RegisterContentFilters();
98  return ptr;
99  });
100  break;
101  } else {
102  continue; // no container, so try the next module
103  }
104  } else {
105  // it must be 'timeout' or 'deferred', which means the module is attempting to process the file
106  // we take this to mean that we stop looking and return the result
107  result = result.then([modulePtr](future<ContainerPtr> fut) {
108  ContainerPtr ptr = fut.get();
109  modulePtr->RegisterContentFilters();
110  return ptr;
111  });
112  break;
113  }
114  }
115 
116  return result;
117 }

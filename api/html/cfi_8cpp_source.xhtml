193 void CFI::AppendComponents(std::stringstream& builder, ComponentList::const_iterator start, ComponentList::const_iterator end)
194 {
195  auto pos = start;
196  while ( pos != end )
197  {
198  builder << "/" << pos->nodeIndex;
199  if ( pos->HasQualifier() )
200  {
201  builder << "[" << pos->qualifier << "]";
202  }
203  if ( pos->HasCharacterOffset() )
204  {
205  builder << ":" << pos->characterOffset;
206 
207  if ( pos->HasTextQualifier() )
208  {
209  builder << "[" << pos->textQualifier << "]";
210  }
211  }
212  else
213  {
214  if ( pos->HasTemporalOffset() )
215  {
216  builder << "~" << pos->temporalOffset;
217  }
218  if ( pos->HasSpatialOffset() )
219  {
220  builder << "@" << pos->spatialOffset.x << ":" << pos->spatialOffset.y;
221  }
222  }
223  if ( pos->IsIndirector() )
224  {
225  builder << "!";
226  }
227 
228  ++pos;
229  }

231 CFI::StringList CFI::CFIComponentStrings(const string &cfi, const string& delimiter)
232 {
233  CFI::StringList components;
234  string breaks = delimiter + "[";
235  string tmp;
236  string::size_type pos = 0, loc = 0;
237 
238  while ( pos < cfi.size() )
239  {
240  loc = cfi.find_first_of(breaks, pos);
241  if ( loc > pos )
242  {
243  if ( loc == string::npos )
244  {
245  tmp.append(cfi, pos, cfi.size()-pos);
246  if ( !tmp.empty() )
247  components.push_back(tmp);
248  tmp.clear();
249  break;
250  }
251  else
252  {
253  tmp.append(cfi, pos, loc-pos);
254  }
255  pos = loc;
256  }
257 
258  if ( cfi[loc] == '[' )
259  {
260  loc = cfi.find_first_of(']', loc);
261  if ( loc == string::npos )
262  {
263  HandleError(EPUBError::CFIParseFailed, _Str("CFI '", cfi, "' has an unterminated qualifier"));
264  }
265 
266  ++loc;
267  tmp.append(cfi, pos, loc-pos);
268  }
269  else if ( cfi.find(delimiter, loc) == loc )
270  {
271  // delimiter found, push the current string
272  if ( !tmp.empty() )
273  components.push_back(tmp);
274  tmp.clear();
275 
276  if ( loc == string::npos )
277  break;
278  loc++;
279  }
280 
281  pos = loc;
282  }
283 
284  if ( !tmp.empty() )
285  components.push_back(tmp);
286 
287  return components;

289 bool CFI::CompileCFI(const string &str)
290 {
291  // strip the 'epubcfi(...)' wrapping
292  string cfi(str);
293  if ( str.find("epubcfi(") == 0 )
294  {
295  cfi = cfi.substr(8, (str.size()-1)-8);
296  }
297  else if ( str.size() == 0 )
298  {
299  HandleError(EPUBError::CFIParseFailed, "Empty CFI string");
300  return false;
301  }
302  else if ( str[0] != '/' )
303  {
304  HandleError(EPUBError::CFINonSlashStartCharacter);
305  }
306 
307  StringList rangePieces = RangedCFIComponents(cfi);
308  if ( rangePieces.size() != 1 && rangePieces.size() != 3 )
309  {
310  HandleError(EPUBError::CFIRangeComponentCountInvalid, _Str("Expected 1 or 3 range components, got ", rangePieces.size()));
311  if ( rangePieces.size() == 0 )
312  return false;
313  }
314 
315  if ( CompileComponentsToList(CFIComponentStrings(rangePieces[0]), &_components) == false )
316  return false;
317 
318  if ( rangePieces.size() >= 3 )
319  {
320  if ( CompileComponentsToList(CFIComponentStrings(rangePieces[1]), &_rangeStart) == false )
321  return false;
322  if ( CompileComponentsToList(CFIComponentStrings(rangePieces[2]), &_rangeEnd) == false )
323  return false;
324 
325  // now sanity-check the range delimiters:
326 
327  // neither should be empty
328  if ( _rangeStart.empty() || _rangeEnd.empty() )
329  {
330  HandleError(EPUBError::CFIRangeInvalid, "One of the supplied range components was empty.");
331  return false;
332  }
333 
334  // check the offsets at the end of each??? they should be the same type
335  if ( (_rangeStart.back().flags & Component::OffsetsMask) != (_rangeEnd.back().flags & Component::OffsetsMask) )
336  {
337  HandleError(EPUBError::CFIRangeInvalid, "Offsets at the end of range components are of different types.");
338  return false;
339  }
340 
341  // ensure that there are no side-bias values
342  if ( (_rangeStart.back().sideBias != SideBias::Unspecified) ||
343  (_rangeEnd.back().sideBias != SideBias::Unspecified) )
344  {
345  HandleError(EPUBError::CFIRangeContainsSideBias);
346  // can safely ignore this one
347  }
348 
349  // where the delimiters' component ranges overlap, start must be <= end
350  auto maxsz = std::max(_rangeStart.size(), _rangeEnd.size());
351  bool inequalNodeIndexFound = false;
352  for ( decltype(maxsz) i = 0; i < maxsz; i++ )
353  {
354  if ( _rangeStart[i].nodeIndex > _rangeEnd[i].nodeIndex )
355  {
356  HandleError(EPUBError::CFIRangeInvalid, "Range components appear to be out of order.");
357  }
358  else if ( !inequalNodeIndexFound && _rangeStart[i].nodeIndex < _rangeEnd[i].nodeIndex )
359  {
360  inequalNodeIndexFound = true;
361  }
362  }
363 
364  // if the two ranges are equal aside from their offsets, the end offset must be > the start offset
365  if ( !inequalNodeIndexFound && _rangeStart.size() == _rangeEnd.size() )
366  {
367  Component &s = _rangeStart.back(), &e = _rangeEnd.back();
368  if ( s.HasCharacterOffset() && s.characterOffset > e.characterOffset )
369  {
370  HandleError(EPUBError::CFIRangeInvalid, "Range components appear to be out of order.");
371  }
372  else
373  {
374  if ( s.HasTemporalOffset() && s.temporalOffset > e.temporalOffset )
375  HandleError(EPUBError::CFIRangeInvalid, "Range components appear to be out of order.");
376  if ( s.HasSpatialOffset() && s.spatialOffset > e.spatialOffset )
377  HandleError(EPUBError::CFIRangeInvalid, "Range components appear to be out of order.");
378  }
379  }
380 
381  _options |= RangeTriplet;
382  }
383 
384  return true;
385 }

416 void CFI::Component::Parse(const string &str)
417 {
418  if ( str.empty() )
419  {
420  HandleError(EPUBError::CFIParseFailed, "Empty string supplied to CFI::Component");
421  return;
422  }
423 
424  std::string utf8 = str.stl_str();
425  std::istringstream iss(utf8);
426 
427  // read an integer
428  iss >> nodeIndex;
429  if ( nodeIndex == 0 && iss.fail() )
430  {
431  HandleError(EPUBError::CFIParseFailed, _Str("No node value at start of CFI::Component string '", str, "'"));
432  return;
433  }
434 
435  while ( !iss.eof() )
436  {
437  char next = 0;
438  iss >> next;
439 
440  switch ( next )
441  {
442  case '[':
443  {
444  size_t pos = static_cast<size_t>(iss.tellg());
445  iss.ignore(std::numeric_limits<std::streamsize>::max(), ']');
446  size_t end = ((size_t)iss.tellg()) - 1;
447 
448  if ( iss.eof() )
449  {
450  HandleError(EPUBError::CFIParseFailed);
451  return;
452  }
453 
454  if ( characterOffset != 0 )
455  {
456  // this is a text qualifier
457  flags |= TextQualifier;
458  std::string sub = utf8.substr(pos, end-pos);
459 
460  // is there a side-bias?
461  auto biasPos = sub.find(";s=");
462  if ( biasPos == std::string::npos )
463  {
464  textQualifier = std::move(sub);
465  }
466  else
467  {
468  textQualifier = sub.substr(0, biasPos);
469  if ( sub.size() > biasPos + 3 )
470  {
471  switch ( sub[biasPos+3] )
472  {
473  case 'b':
474  sideBias = SideBias::Before;
475  break;
476  case 'a':
477  sideBias = SideBias::After;
478  break;
479  default:
480  sideBias = SideBias::Unspecified;
481  break;
482  }
483  }
484  }
485  }
486  else
487  {
488  // it's a position qualifier
489  qualifier = utf8.substr(pos, end-pos);
490  flags |= Qualifier;
491  }
492 
493  break;
494  }
495 
496  case '~':
497  {
498  // character offsets and spatial/temporal offsets are mutually exclusive
499  if ( HasCharacterOffset() )
500  break;
501 
502  // read a numeral
503  iss >> temporalOffset;
504  flags |= TemporalOffset;
505  break;
506  }
507 
508  case '@':
509  {
510  // character offsets and spatial/temporal offsets are mutually exclusive
511  if ( HasCharacterOffset() )
512  break;
513 
514  // two floats, separated by a colon
515  float x, y;
516 
517  // read x
518  iss >> x;
519 
520  // check for and skip delimiter
521  if ( iss.peek() != ':' )
522  break;
523  iss.ignore(1);
524 
525  // read y
526  iss >> y;
527 
528  spatialOffset.x = x;
529  spatialOffset.y = y;
530  flags |= SpatialOffset;
531  break;
532  }
533 
534  case ':':
535  {
536  // character offsets and spatial/temporal offsets are mutually exclusive
537  if ( HasSpatialTemporalOffset() )
538  break;
539 
540  iss >> characterOffset;
541  flags |= CharacterOffset;
542  break;
543  }
544 
545  case '!':
546  {
547  // must be the last character, and no offsets
548  if ( ((int)iss.peek()) != -1 || HasSpatialTemporalOffset() || HasCharacterOffset() )
549  break;
550 
551  flags |= Indirector;
552  break;
553  }
554 
555  default:
556  break;
557  }
558  }

561 {

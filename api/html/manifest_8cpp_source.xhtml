67 ItemProperties& ItemProperties::operator=(const string& attrStr)
68 {
69  if ( attrStr.empty() )
70  {
71  _p = None;
72  return *this;
73  }
74 
75  // ensure the attributes string is lowercase
76  string lowAttrs = attrStr.tolower();
77 
78  // NB: this is a C++11 raw-string literal. R"" means 'raw string', and the X(...)X bit are delimiters.
79  REGEX_NS::regex re("\\w+(-\\w+)?", REGEX_NS::regex::icase);
80  auto pos = REGEX_NS::sregex_iterator(lowAttrs.stl_str().begin(), lowAttrs.stl_str().end(), re);
81  auto end = REGEX_NS::sregex_iterator();
82 
83  for ( ; pos != end; pos++ )
84  {
85  // using the entire matched range
86  auto found = PropertyLookupTable.find(pos->str());
87  if ( found != PropertyLookupTable.end() )
88  {
89  _p |= found->second;
90  }
91  else
92  {
93  printf("Property not found: %s (from %s)\n", pos->str().c_str(), attrStr.c_str());
94  }
95  }
96 
97  return *this;
98 }
99 string ItemProperties::str() const
100 {
101  if ( _p == None )
102  return "";
103 
104  std::vector<string> vec;
105  value_type test = 1;
106 
107  while ( test < AllPropertiesMask )
108  {
109  if ( (_p & test) == test )
110  {
111  switch ( test )
112  {
113  case CoverImage:
114  vec.emplace_back("cover-image");
115  break;
116  case ContainsMathML:
117  vec.emplace_back("mathml");
118  break;
119  case Navigation:
120  vec.emplace_back("nav");
121  break;
122  case HasRemoteResources:
123  vec.emplace_back("remote-resources");
124  break;
125  case HasScriptedContent:
126  vec.emplace_back("scripted");
127  break;
128  case ContainsSVG:
129  vec.emplace_back("svg");
130  break;
131  case ContainsSwitch:
132  vec.emplace_back("switch");
133  break;
134  default:
135  break;
136  }
137  }
138 
139  test <<= 1;
140  }
141 
142  if ( vec.size() == 0 )
143  return "";
144 
145  std::stringstream builder;
146  builder << vec[0];
147  for ( decltype(vec.size()) i = 0; i < vec.size(); i++ )
148  {
149  builder << ", " << vec[i];
150  }
151 
152  return builder.str();
153 }

237 shared_ptr<xml::Document> ManifestItem::ReferencedDocument() const
238 {
239  // TODO: handle remote URLs
240  string path(BaseHref());
241 
242  auto package = this->Owner();
243  if ( !package )
244  return nullptr;
245 
246  unique_ptr<ArchiveXmlReader> reader = package->XmlReaderForRelativePath(path);
247  if ( !reader )
248  return nullptr;
249 
250  // In some EPUBs, UTF-8 XML/HTML files have a superfluous (erroneous?) BOM, so we either:
251  // pass "utf-8" and expect InputBuffer::read_cb (in io.cpp) to skip the 3 erroneous bytes
252  // (otherwise the XML parser fails),
253  // or we pass NULL (in which case the parser auto-detects encoding)
254  const char * encoding = nullptr;
255  //const char * encoding = "utf-8";
256 
257  shared_ptr<xml::Document> result(nullptr);
258 #if EPUB_USE(LIBXML2)
259  int flags = XML_PARSE_RECOVER|XML_PARSE_NOENT|XML_PARSE_DTDATTR;
260  if ( _mediaType == "text/html" )
261  result = reader->htmlReadDocument(path.c_str(), encoding, flags);
262  else
263  result = reader->xmlReadDocument(path.c_str(), encoding, flags);
264 #elif EPUB_USE(WIN_XML)
265  result = reader->ReadDocument(path.c_str(), "utf-8", 0);
266 #endif
267  return result;
268 }

583 void Node::Wrap(_xmlNode *aNode)
584 {
585  void* wrapper = nullptr;
586  switch ( aNode->type )
587  {
588  case XML_DOCUMENT_NODE:
589  case XML_DOCUMENT_FRAG_NODE:
590  case XML_HTML_DOCUMENT_NODE:
591  {
592  wrapper = new LibXML2Private<class Document>(new class Document(reinterpret_cast<xmlDocPtr>(aNode)));
593  break;
594  }
595 
596  case XML_DTD_NODE:
597  wrapper = new LibXML2Private<DTD>(new DTD(reinterpret_cast<xmlDtdPtr>(aNode)));
598  break;
599 
600  case XML_NAMESPACE_DECL:
601  wrapper = new LibXML2Private<class Namespace>(new class Namespace(reinterpret_cast<xmlNsPtr>(aNode)));
602  break;
603 
604  case XML_ATTRIBUTE_NODE:
605  //wrapper = new Attribute(reinterpret_cast<xmlAttrPtr>(aNode));
606  break;
607 
608  case XML_ELEMENT_NODE:
609  wrapper = new LibXML2Private<Element>(new Element(aNode));
610  break;
611 
612  default:
613  wrapper = new LibXML2Private<Node>(new Node(aNode));
614  break;
615  }
616 
617  // The _private ptr in the xmlNodePtr is a POINTER TO a LibXML2Private
618  // object. This means that the allocated object is shared, and the
619  // xmlNodePtr holds a strong reference, released when the xmlNodePtr
620  // is deallocated.
621  aNode->_private = wrapper;
622 }
623 void Node::Unwrap(_xmlNode *aNode)
624 {
625  typedef LibXML2Private<class Namespace> NsPrivate;
626  typedef LibXML2Private<Node> NodePrivate;
627 
628  if (aNode->type == XML_NAMESPACE_DECL)
629  {
630  // _xmlNs is laid out differently -- _private is in a different place...
631  xmlNsPtr __ns = reinterpret_cast<xmlNsPtr>(aNode);
632  if (__ns->_private != nullptr)
633  {
634  NsPrivate* ptr = reinterpret_cast<NsPrivate*>(__ns->_private);
635  if (ptr->__sig == _READIUM_XML_SIGNATURE)
636  {
637  ptr->__ptr->release();
638  delete ptr;
639  }
640  __ns->_private = nullptr;
641  }
642  }
643  else if (aNode->_private != nullptr)
644  {
645  NodePrivate* ptr = reinterpret_cast<NodePrivate*>(aNode->_private);
646  if (ptr->__sig == _READIUM_XML_SIGNATURE)

674 void Node::rebind(_xmlNode *newNode)
675 {
676  if ( _xml == newNode )
677  return;
678 
679  typedef LibXML2Private<Node> _Private;
680 
681  if ( _xml != nullptr )
682  {
683  if ( _xml->parent == nullptr && _xml->next == nullptr && _xml->prev == nullptr )
684  {
685  xmlNodePtr __n = _xml;
686  this->release();
687  xmlFreeNode(__n); // releases libxml's reference to this object
688  }
689  }
690 
691  _xml = newNode;
692 
693  typedef std::shared_ptr<Node> NodePtr;
694  if (_xml->_private != nullptr && *((unsigned int*)(_xml->_private)) == _READIUM_XML_SIGNATURE)
695  {
696  // reassign the bridge ptr on the libxml node
697  _Private* ptr = reinterpret_cast<_Private*>(newNode->_private);
698  ptr->__ptr = shared_from_this();
699  }
700  else if (_xml->_private != nullptr)
701  {
702  auto ptr = shared_from_this();
703  _xml->_private = new _Private(ptr);
704  }
705 }

41 std::shared_ptr<AsyncByteStream> FilterChain::GetFilteredOutputStreamForManifestItem(ConstManifestItemPtr item) const
42 {
43  std::unique_ptr<AsyncByteStream> rawInput = item->AsyncReader();
44 
45  // transfer ownership to a new shared_ptr
46  ChainLink input = ChainLink(rawInput.release());
47  shared_vector<ChainLinkProcessor> thisChain;
48 
49  AsyncPipe::Pair linkPipe;
50 
51  for ( ContentFilterPtr filter : _filters )
52  {
53  if ( filter->TypeSniffer()(item) )
54  {
55  if ( !thisChain.empty() )
56  thisChain.back()->SetOutputLink(linkPipe.first);
57 
58  thisChain.push_back(ChainLinkProcessor::New(filter, input, item));
59  linkPipe = AsyncPipe::LinkedPair();
60  input = linkPipe.second;
61  }
62  }
63 
64  // if no filters apply, read raw bytes
65  if ( thisChain.empty() )
66  return input;
67 
68  // otherwise, attach the output pipe
69  thisChain.back()->SetOutputLink(linkPipe.first);
70 
71  static std::once_flag __once;
72  std::call_once(__once, [](){
73  _filterThreadPool.reset(new thread_pool(thread_pool::Automatic));
74  });
75 
76  // set it to run on a runloop somewhere...
77  _filterThreadPool->add([thisChain]() mutable {
78  RunLoopPtr myRunLoop = RunLoop::CurrentRunLoop();
79 
80  auto pos = thisChain.rbegin();
81  auto end = thisChain.rend();
82  while ( pos != end )
83  {
84  (*pos)->ScheduleProcessor(myRunLoop);
85  ++pos;
86  }
87 
88  // as a file/zip-based async stream, the raw input stream will automatically
89  // read bytes from the underlying resource on a shared background thread,
90  // thus feeding the chain that follows
91 
92  // Note that the input stream is returned to us already opened.
93 
94  // run until all sources have been removed
95  myRunLoop->Run();
96  });
97 
98  // return the output pipe
99  return linkPipe.second;
100 }

138 std::unique_ptr<ByteStream> FilterChain::GetFilterChainByteStreamRange(ConstManifestItemPtr item, SeekableByteStream *rawInput) const
139 {
140  unique_ptr<ByteStream> resultStream;
141  uint nFilters = 0;
142  for (ContentFilterPtr filter : _filters)
143  {
144  if (filter->TypeSniffer()(item))
145  {
146  nFilters++;
147  if (nFilters > 1)
148  {
149  continue;
150  }
151 
152  if (filter->GetOperatingMode() == ContentFilter::OperatingMode::SupportsByteRanges)
153  {
154  unique_ptr<SeekableByteStream> rawInputPtr(rawInput);
155  resultStream.reset(new FilterChainByteStreamRange(std::move(rawInputPtr), filter, item));
156  }
157  }
158  }
159 
160  if (nFilters > 1)
161  {
162  // more than one filter...abort!
163  return nullptr;
164  }
165 
166  // There are no ContentFilter classes that curretly apply.
167  // In this case, return an empty FilterChainByteStreamRange, that will simply put out raw bytes.
168  if (!resultStream)
169  {
170  unique_ptr<SeekableByteStream> rawInputPtr(rawInput);
171  resultStream.reset(new FilterChainByteStreamRange(std::move(rawInputPtr), nullptr, nullptr));
172  }
173 
174  return resultStream;
175 }

206 void FilterChain::ChainLinkProcessor::ScheduleProcessor(RunLoopPtr runLoop)
207 {
208  if ( _filter->RequiresCompleteData() )
209  _collectionBuffer.SetUsesSecureErasure();
210 
211  std::weak_ptr<ChainLinkProcessor> weakSelf(Ptr());
212  _input->SetEventHandler([this, weakSelf](AsyncEvent evt, AsyncByteStream* stream) {
213  // we use this variable ONLY to ensure that the 'this' ptr is still valid
214  auto strongSelf = weakSelf.lock();
215  if ( !bool(strongSelf) )
216  return;
217 
218  switch ( evt )
219  {
220  case AsyncEvent::HasBytesAvailable:
221  {
222  // we *know* that all links on a given chain fire events serially on
223  // the same thread, so we don't need to dick about with locks here
224 
225  // there are bytes available to read -- pump as many through as we can
226  ssize_t numMoved = FunnelBytes();
227  if ( numMoved == 0 && !_output->IsOpen() )
228  stream->Close();
229 
230  break;
231  }
232 
233  case AsyncEvent::ErrorOccurred:
234  std::cerr << "ChainLinkProcessor input stream error: " << stream->Error() << std::endl;
235  _output->Close();
236  break;
237 
238  case AsyncEvent::EndEncountered:
239  {
240  if ( _filter->RequiresCompleteData() )
241  {
242  size_t filteredLen = 0;
243  void* filteredData = _filter->FilterData(_context.get(), _collectionBuffer.GetBytes(), _collectionBuffer.GetBufferSize(), &filteredLen);
244  if ( filteredData == nullptr || filteredLen == 0 ) {
245  if (filteredData != nullptr && filteredData != _collectionBuffer.GetBytes())
246  delete[] reinterpret_cast<uint8_t*>(filteredData);
247  throw std::logic_error("ChainLinkProcessor: ContentFilter::FilterData() returned no data!");
248  }
249 
250  // forward the data
251  uint8_t *outputData = reinterpret_cast<uint8_t*>(filteredData);
252  size_t offset = 0;
253 
254  while ( offset < filteredLen )
255  {
256  offset += _output->WriteBytes(&outputData[offset], filteredLen-offset);
257  if ( offset < filteredLen )
258  std::this_thread::yield();
259  }
260 
261  if (filteredData != _collectionBuffer.GetBytes())
262  delete[] reinterpret_cast<uint8_t*>(filteredData);
263 
264  // (securely) remove all bytes from the buffer
265  _collectionBuffer.RemoveBytes(_collectionBuffer.GetBufferSize());
266  }
267 
268  _output->Close();
269  break;
270  }
271  default:
272  break;
273  }
274  });
275 
276  // if this is the last link in the chain, this will ultimately be replaced by the
277  // SDK client. If it's only an intermediary link, however, we need to ensure that
278  // notification of read-space-available messages is passed back down the chain to
279  // the actual resource stream, otherwise the whole process may stall...
280  _output->SetEventHandler([this](AsyncEvent evt, AsyncByteStream* stream) {
281  switch ( evt )
282  {
283  case AsyncEvent::HasSpaceAvailable:
284  {
285  // pull more data through into the output stream
286  if ( FunnelBytes() <= 0 && (_input->AtEnd() || !_input->IsOpen()) )
287  {
288  stream->Close();
289  return;
290  }
291 
292  break;
293  }
294 
295  case AsyncEvent::ErrorOccurred:
296  std::cerr << "ChainLinkProcessor input stream error: " << stream->Error() << std::endl;
297  stream->Close();
298  if (bool(_input))
299  _input->Close();
300  break;
301 
302  case AsyncEvent::EndEncountered:
303  if (bool(_input))
304  _input->Close();
305  break;
306 
307  default:
308  break;
309  }
310  });
311 
312  _input->SetTargetRunLoop(runLoop);
313  _output->SetTargetRunLoop(runLoop);
314 }

316 ssize_t FilterChain::ChainLinkProcessor::FunnelBytes()
317 {
318  uint8_t buf[ASYNC_BUF_SIZE];
319  size_t bytesToMove = std::min(_input->BytesAvailable(), _output->SpaceAvailable());
320 
321  while ( bytesToMove > 0 )
322  {
323  size_t thisChunk = std::min(size_t(ASYNC_BUF_SIZE), bytesToMove);
324  thisChunk = _input->ReadBytes(buf, thisChunk); // consumes read bytes from the buffer
325 
326  if ( _filter->RequiresCompleteData() )
327  {
328  _collectionBuffer.AddBytes(buf, thisChunk);
329  }
330  else
331  {
332  size_t filteredLen = 0;
333  void* filteredData = _filter->FilterData(_context.get(), buf, thisChunk, &filteredLen);
334  if ( filteredData == nullptr || filteredLen == 0 ) {
335  if (filteredData != nullptr && filteredData != buf)
336  delete[] reinterpret_cast<uint8_t*>(filteredData);
337  throw std::logic_error("ChainLinkProcessor: ContentFilter::FilterData() returned no data!");
338  }
339 
340  _output->WriteBytes(filteredData, filteredLen);
341 
342  if (filteredData != buf)
343  delete[] reinterpret_cast<uint8_t*>(filteredData);
344  }
345 
346  bytesToMove -= thisChunk;
347  }
348 
349  return (ssize_t)bytesToMove;
350 }

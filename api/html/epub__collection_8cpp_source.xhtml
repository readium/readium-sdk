40 bool Collection::ParseXML(shared_ptr<xml::Node> node)
41 {
42  try
43  {
44  SetXMLIdentifier(_getProp(node, "id"));
45 
46  _role = _getProp(node, "role", ePub3NamespaceURI);
47  if (_role.empty())
48  HandleError(EPUBError::OPFCollectionMissingRole);
49 
50  size_t nodeIdx = 0;
51  for (auto child = node->FirstElementChild(); bool(child); child = child->NextElementSibling(), ++nodeIdx)
52  {
53  string name(child->Name());
54 
55  if (name == "metadata")
56  {
57  if (nodeIdx != 0)
58  HandleError(EPUBError::OPFCollectionMetadataOutOfOrder);
59 
60  // parse the metadata elements
61  ParseMetadata(child);
62  }
63  else if (name == "collection")
64  {
65  if (_links.size() != 0 )
66  HandleError(EPUBError::OPFCollectionSubcollectionOutOfOrder);
67 
68  CollectionPtr sub = New(Owner(), shared_from_this());
69  if (sub->ParseXML(child))
70  {
71 #if EPUB_HAVE(CXX_MAP_EMPLACE)
72  _childCollections.emplace(sub->Role(), sub);
73 #else
74  _childCollections[sub->Role()] == sub;
75 #endif
76  }
77  }
78  else if (name == "link")
79  {
80  LinkPtr link = Link::New(shared_from_this());
81  if (link->ParseXML(child))
82  _links.push_back(link);
83  }
84  }
85 
86  if (_links.empty())
87  HandleError(EPUBError::OPFCollectionNoLinks);
88  }
89  catch (const epub_spec_error&)
90  {
91  return false;
92  }
93 
94  return true;
95 }
96 void Collection::ParseMetadata(shared_ptr<xml::Node> node)
97 {
98  PropertyHolderPtr holderPtr = CastPtr<PropertyHolder>();
99  for (auto metaNode = node->FirstElementChild(); bool(metaNode); metaNode = metaNode->NextElementSibling())
100  {
101  if (!_getProp(metaNode, "refines").empty())
102  {
103  HandleError(EPUBError::OPFCollectionLinkIncludesRefinement);
104  continue;
105  }
106 
107  if (metaNode->Name() == "meta")
108  {
109  HandleError(EPUBError::OPFCollectionIllegalMetaElement);
110  continue;
111  }
112 
113  PropertyPtr p = Property::New(holderPtr);
114  if (p->ParseMetaElement(metaNode))
115  AddProperty(p);
116  }
117 }

48 ObjectPreprocessor::ObjectPreprocessor(ConstPackagePtr pkg, const string& buttonTitle) : ContentFilter(ShouldApply), _button(buttonTitle)
49 {
50  Package::StringList mediaTypes = pkg->MediaTypesWithDHTMLHandlers();
51  if ( mediaTypes.empty() )
52  {
53  // No work for the filter to do here -- disable all matches
54  SetTypeSniffer([](ConstManifestItemPtr){return false;});
55  return;
56  }
57 
58  std::stringstream ss;
59  ss << "<object\\s+?([^>]*?(?:media-)?type=\"(";
60 
61  auto pos = mediaTypes.begin();
62  auto end = mediaTypes.end();
63 
64  while ( pos != end )
65  {
66  auto here = pos++;
67  std::string regstr("$`\\\\$&$'");
68  std::string str = REGEX_NS::regex_replace(here->stl_str(), reEscaper, regstr); // yes, double-backslash, so it doesn't escape whatever '$&' is (i.e. the character which needs to be escaped)
69  if ( pos == end )
70  ss << str;
71  else
72  ss << str << "|";
73  }
74 
75  ss << ")\"[^>]*?)>((?:.|\\n|\\r)*?)</object>";
76  std::string reStr = ss.str();
77  _objectMatcher = REGEX_NS::regex(reStr, regexFlags);
78 
79  for ( auto mediaType : mediaTypes )
80  {
81 #if EPUB_HAVE(CXX_MAP_EMPLACE)
82  _handlers.emplace(mediaType, *(pkg->OPFHandlerForMediaType(mediaType)));
83 #else
84  _handlers.insert({mediaType, *(pkg->OPFHandlerForMediaType(mediaType))});
85 #endif
86  }
87 }
88 void* ObjectPreprocessor::FilterData(FilterContext* context, void *data, size_t len, size_t *outputLen)
89 {
90  char* input = reinterpret_cast<char*>(data);
91  // find each `object` tag
92  REGEX_NS::cregex_iterator pos(input, input+len, _objectMatcher);
93  REGEX_NS::cregex_iterator end;
94  if ( pos == end )
95  {
96  *outputLen = len;
97  return data; // no match == no change
98  }
99 
100  std::string output;
101  while ( pos != end )
102  {
103  // output any leading non-matched characters
104  output += pos->prefix();
105 
106  std::string type(pos->str(2));
107  // we have matched an <object> element: find the appropriate media handler
108  auto found = _handlers.find(type);
109  if ( found == _handlers.end() )
110  {
111  output += pos->str();
112  continue;
113  }
114 
115  const MediaHandler& handler = found->second;
116  ContentHandler::ParameterList params;
117  params["type"] = type;
118 
119  // find the data source
120  REGEX_NS::smatch m;
121  std::string attrs(pos->str(1));
122  REGEX_NS::regex_search(attrs.cbegin(), attrs.cend(), m, SourceFinder);
123  string src(m[1].str());
124 
125  // find any parameters to the object tag
126  std::string content(pos->str(3));
127  REGEX_NS::sregex_iterator cpos(content.begin(), content.end(), ParamMatcher);
128  REGEX_NS::sregex_iterator cend;
129 
130  while ( cpos != cend )
131  {
132  if ( cpos->length() >= 4 )
133  {
134  string name, value;
135  if ( cpos->str(1) == "name" )
136  name = cpos->str(2);
137  else
138  value = cpos->str(2);
139 
140  if ( cpos->str(3) == "value" )
141  value = cpos->str(4);
142  else
143  name = cpos->str(4);
144 
145  params[name] = value;
146  }
147 
148  ++cpos;
149  }
150 
151  // now determine the target-- this is an absolute URL
152  IRI target = handler.Target(src, params);
153 
154  // find out if the object tag had an id attribute
155  std::string objectID;
156  if ( REGEX_NS::regex_search(attrs.cbegin(), attrs.cend(), m, IDFinder) )
157  objectID = m[1].str();
158 
159  // now construct the `iframe` tag
160  std::string url = target.URIString().stl_str();
161  output += "<iframe src=\"" + url + "\" srcdoc=\"" + url + "\"";
162 
163  // replicate any id attribute from the `object` tag
164  if ( !objectID.empty() )
165  output += " id=\"" + objectID + "\"";
166 
167  // enable sandbox and allow some stuff, and use seamless presentation
168  output += " sandbox=\"allow-forms allow-scripts allow-same-origin\" seamless=\"seamless\"></iframe>";
169 
170  // now add the form & button
171  output += "<form action=\"" + url + "\" method=\"get\"";
172  if ( !objectID.empty() )
173  output += " id=\"" + objectID + "-form\"";
174  output += "><button type=\"submit\"";
175  if ( !objectID.empty() )
176  output += " id=\"" + objectID + "-button\"";
177  output += ">" + _button.stl_str() + "</button></form>";
178 
179  // that's it-- we've replaced the whole lot!
180 
181  // step forward, and if we get the end let's output the current suffix
182  // this ensures that we don't lose everything following the last match
183  auto here = pos++;
184  if ( pos == end )
185  output += here->suffix();
186  }
187 
188  *outputLen = output.size();
189  if ( output.size() < len )
190  {
191  // use the incoming buffer directly
192  output.copy(input, output.size());
193  return input;
194  }
195 
196  // allocate an output buffer
197  char* result = new char[output.size()];
198  output.copy(result, output.size());
199  return result;
200 }

464 bool Package::Unpack()
465 {
466  PackagePtr sharedMe = shared_from_this();
467 
468  // very basic sanity check
469  auto root = _opf->Root();
470  string rootName(root->Name());
471  rootName.tolower();
472  bool isEPUB3 = true;
473  string versionStr;
474  int version = 0;
475 
476 
477  if ( rootName != "package" )
478  {
479  HandleError(EPUBError::OPFInvalidPackageDocument);
480  return false; // not an OPF file, innit?
481  }
482  versionStr = _getProp(root, "version");
483  if ( versionStr.empty() )
484  {
485  HandleError(EPUBError::OPFPackageHasNoVersion);
486  }
487  else
488  {
489  // GNU libstdc++ seems to not want to let us use these C++11 routines...
490 #ifndef _LIBCPP_VERSION
491  version = (int)strtol(versionStr.c_str(), nullptr, 10);
492 #else
493  version = std::stoi(versionStr.stl_str());
494 #endif
495 
496  if (version < 3)
497  isEPUB3 = false;
498  }
499 
500  auto val = _getProp(root, "prefix", ePub3NamespaceURI);
501  InstallPrefixesFromAttributeValue(val);
502 
503  // go through children to determine the CFI index of the <spine> tag
504  static xml::string kSpineName((const char*)"spine");
505  static xml::string kManifestName((const char*)"manifest");
506  static xml::string kMetadataName((const char*)"metadata");
507 
508  _spineCFIIndex = 0;
509  uint32_t idx = 0;
510  auto child = root->FirstElementChild();
511  while ( bool(child) )
512  {
513  idx += 2;
514  if ( child->Name() == kSpineName )
515  {
516  _spineCFIIndex = idx;
517  if ( _spineCFIIndex != 6 )
518  HandleError(EPUBError::OPFSpineOutOfOrder);
519  }
520  else if ( child->Name() == kManifestName && idx != 4 )
521  {
522  HandleError(EPUBError::OPFManifestOutOfOrder);
523  }
524  else if ( child->Name() == kMetadataName && idx != 2 )
525  {
526  HandleError(EPUBError::OPFMetadataOutOfOrder);
527  }
528 
529  child = child->NextElementSibling();
530  }
531 
532  if ( _spineCFIIndex == 0 )
533  {
534  HandleError(EPUBError::OPFNoSpine);
535  return false; // spineless!
536  }
537 
538 #if EPUB_COMPILER_SUPPORTS(CXX_INITIALIZER_LISTS)
539  XPathWrangler xpath(_opf, {{"opf", OPFNamespace}, {"dc", DCNamespace}});
540 #else
541  XPathWrangler::NamespaceList __m;
542  __m["opf"] = OPFNamespace;
543  __m["dc"] = DCNamespace;
544  XPathWrangler xpath(_opf, __m);
545 #endif
546 
547  // simple things: manifest and spine items
548  xml::NodeSet manifestNodes;
549  xml::NodeSet spineNodes;
550 
551  try
552  {
553  manifestNodes = xpath.Nodes("/opf:package/opf:manifest/opf:item");
554  spineNodes = xpath.Nodes("/opf:package/opf:spine/opf:itemref");
555 
556  if ( manifestNodes.empty() )
557  {
558  HandleError(EPUBError::OPFNoManifestItems);
559  }
560 
561  if ( spineNodes.empty() )
562  {
563  HandleError(EPUBError::OPFNoSpineItems);
564  }
565 
566  for ( auto node : manifestNodes )
567  {
568  auto p = ManifestItem::New(sharedMe);
569  if ( p->ParseXML(node) )
570  {
571 #if EPUB_HAVE(CXX_MAP_EMPLACE)
572  _manifest.emplace(p->Identifier(), p);
573 #else
574  _manifest[p->Identifier()] = p;
575 #endif
576  StoreXMLIdentifiable(p);
577  }
578  else
579  {
580  // TODO: Need an error here
581  }
582  }
583 
584  // check fallback chains
585  typedef std::map<string, bool> IdentSet;
586  IdentSet idents;
587  for ( auto &pair : _manifest )
588  {
589  ManifestItemPtr item = pair.second;
590  if ( item->FallbackID().empty() )
591  continue;
592 
593  idents[item->XMLIdentifier()] = true;
594  while ( item != nullptr && !item->FallbackID().empty() )
595  {
596  if ( idents[item->FallbackID()] )
597  {
598  HandleError(EPUBError::OPFFallbackChainCircularReference);
599  break;
600  }
601 
602  item = item->Fallback();
603  }
604 
605  idents.clear();
606  }
607 
608  SpineItemPtr cur;
609  for ( auto node : spineNodes )
610  {
611  auto next = SpineItem::New(sharedMe);
612  if ( next->ParseXML(node) == false )
613  {
614  // TODO: need an error code here
615  continue;
616  }
617 
618  // validation of idref
619  auto manifestFound = _manifest.find(next->Idref());
620  if ( manifestFound == _manifest.end() )
621  {
622  HandleError(EPUBError::OPFInvalidSpineIdref, _Str(next->Idref(), " does not correspond to a manifest item"));
623  continue;
624  }
625 
626  // validation of spine resource type w/fallbacks
627  ManifestItemPtr manifestItem = next->ManifestItem();
628  bool isContentDoc = false;
629  do
630  {
631  if ( manifestItem->MediaType() == "application/xhtml+xml" ||
632  manifestItem->MediaType() == "image/svg" )
633  {
634  isContentDoc = true;
635  break;
636  }
637 
638  } while ( (manifestItem = manifestItem->Fallback()) );
639 
640  if ( !isContentDoc )
641  HandleError(EPUBError::OPFFallbackChainHasNoContentDocument);
642 
643  StoreXMLIdentifiable(next);
644 
645  if ( cur != nullptr )
646  {
647  cur->SetNextItem(next);
648  }
649  else
650  {
651  _spine = next;
652  }
653 
654  cur = next;
655  }

150 void XPathEvaluator::PerformFunction(xmlXPathParserContextPtr ctx, const string &name, const string &uri, int nargs)
151 {
152  string key;
153  if ( uri.length() > 0 )
154  key = uri + "::" + name;
155  else
156  key = name;
157 
158  auto iter = _functions.find(key);
159  if ( iter == _functions.end() )
160  {
161  // eat the stack & push false
162  for ( int i = 0; i < nargs; i++ )
163  valuePop(ctx);
164  valuePush(ctx, xmlXPathNewBoolean(0));
165  return;
166  }
167 
168  // call the function
169  iter->second(ctx, nargs);
170 }
171 bool XPathEvaluator::RegisterVariable(const string &name, void *data, ObjectType type, const string &namespaceURI)
172 {
173  xmlXPathObjectPtr xValue = nullptr;
174  switch ( type )
175  {
176  case ObjectType::Undefined:
177  default:
178  return false;
179 
180  case ObjectType::NodeSet:
181  {
182  const xml::NodeSet * nodes = reinterpret_cast<const xml::NodeSet*>(data);
183  if ( typeid(xml::NodeSet) != typeid(nodes) )
184  return false;
185 
186  xmlNodePtr * ptr = reinterpret_cast<xmlNodePtr*>(xmlMemMalloc(sizeof(xmlNodePtr) * nodes->size()));
187  if ( ptr == nullptr )
188  return false;
189 
190  xmlNodeSetPtr set = xmlXPathNodeSetCreate(nullptr);
191  if ( set == nullptr )
192  return false;
193 
194  for ( size_t i = 0; i < nodes->size(); i++ )
195  xmlXPathNodeSetAdd(set, nodes->at(i)->xml());
196 
197  xValue = xmlXPathNewNodeSetList(set);
198  break;
199  }
200 
201  case ObjectType::Boolean:
202  {
203  const bool * val = reinterpret_cast<const bool *>(data);
204  if ( typeid(bool) != typeid(val) )
205  return false;
206  xValue = xmlXPathNewBoolean((*val ? 1 : 0));
207  break;
208  }
209 
210  case ObjectType::Number:
211  {
212  const double * val = reinterpret_cast<const double *>(data);
213  xValue = xmlXPathNewFloat(*val);
214  break;
215  }
216 
217  case ObjectType::String:
218  {
219  const string * str = reinterpret_cast<const string*>(data);
220  if ( typeid(string) == typeid(str) )
221  {
222  xValue = xmlXPathNewString(str->utf8());
223  break;
224  }
225 
226  const xmlChar * chars = reinterpret_cast<const xmlChar *>(data);
227  xValue = xmlXPathNewString(chars);
228  break;
229  }
230  }
231 
232  if ( namespaceURI.length() == 0 )
233  return ( xmlXPathRegisterVariable(_ctx, name.utf8(), xValue) == 0 );
234 
235  return ( xmlXPathRegisterVariableNS(_ctx, name.utf8(), namespaceURI.utf8(), xValue) == 0 );
236 }

290 ePub3::xml::NodeSet XPathEvaluator::NodeSetResult() const
291 {

295  throw std::domain_error(std::string(__PRETTY_FUNCTION__) + " called when last result is not a Node Set");
296

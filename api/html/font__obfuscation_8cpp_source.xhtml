54 void * FontObfuscator::FilterData(FilterContext* context, void *data, size_t len, size_t *outputLen)
55 {
56  FontObfuscationContext* p = dynamic_cast<FontObfuscationContext*>(context);
57  size_t bytesFiltered = p->ProcessedCount();
58 
59  uint8_t *buf = static_cast<uint8_t*>(data);
60  for ( size_t i = 0; i < len && (i + bytesFiltered) < 1040; i++)
61  {
62  // XOR each of the first 1040 bytes of the font with the key, circling around the keybuf
63  buf[i] ^= _key[(i+bytesFiltered)%20];
64  }
65 
66  bytesFiltered += len;
67  p->SetProcessedCount(bytesFiltered);
68  *outputLen = len;
69  return buf;
70 }
71 bool FontObfuscator::BuildKey(ConstContainerPtr container)
72 {
73  REGEX_NS::regex re("\\s+");
74  std::stringstream ss;
75 
76  for ( auto pkg : container->Packages() )
77  {
78  if ( ss.tellp() > 0 )
79  ss << ' ';
80 
81  // we use a C++11 regex to remove all whitespace in the value
82  std::string replacement;
83  ss << REGEX_NS::regex_replace(pkg->PackageID().stl_str(), re, replacement);
84  }
85 
86  auto str = ss.str();
87 
88 #if EPUB_PLATFORM(WIN)
89  HCRYPTPROV csp;
90  if ( ::CryptAcquireContext(&csp, NULL, NULL, PROV_DSS, CRYPT_VERIFYCONTEXT) == FALSE )
91  {
92  _THROW_LAST_ERROR_();
93  }
94 
95  HCRYPTHASH hasher;
96  if ( ::CryptCreateHash(csp, CALG_SHA, 0, 0, &hasher) == FALSE )
97  {
98  ::CryptReleaseContext(csp, 0);
99  _THROW_LAST_ERROR_();
100  }
101 
102  DWORD winerr = NO_ERROR;
103  if ( ::CryptHashData(hasher, reinterpret_cast<const BYTE*>(str.data()), str.length(), 0) == TRUE )
104  {
105  DWORD len = KeySize;
106  if ( ::CryptGetHashParam(hasher, HP_HASHVAL, _key, &len, 0) == FALSE )
107  winerr = ::GetLastError();
108  }
109  else
110  {
111  winerr = ::GetLastError();
112  }
113 
114  ::CryptDestroyHash(hasher);
115  ::CryptReleaseContext(csp, 0);
116 
117  if ( winerr != NO_ERROR )
118  _THROW_WIN_ERROR_(winerr);
119 #elif EPUB_PLATFORM(WINRT)
120  auto byteArray = ArrayReference<byte>(reinterpret_cast<byte*>(const_cast<char*>(str.data())), str.length());
121  auto inBuf = CryptographicBuffer::CreateFromByteArray(byteArray);
122  auto keyBuf = HashAlgorithmProvider::OpenAlgorithm(HashAlgorithmNames::Sha1)->HashData(inBuf);
123 
124  Array<byte>^ outArray = nullptr;
125  CryptographicBuffer::CopyToByteArray(keyBuf, &outArray); // creates a new Array<byte>^ and returns it by reference
126  memcpy_s(_key, KeySize, outArray->Data, outArray->Length);
127 #else
128  // hash the accumulated string (using OpenSSL syntax for portability)
129  SHA_CTX ctx;
130  SHA1_Init(&ctx);
131  SHA1_Update(&ctx, str.data(), str.length());
132  SHA1_Final(_key, &ctx);
133 #endif
134  return true;
135 }
